1function [k] = k1(material,geom,L)
 2%k_lin_beam
 3%   Calculate the local element stiffness matrix.  3D beam element, Cook et al.
 4%   (2002)
 5
 6% Extract variables
 7E = material(1);
 8nu = material(2);
 9G = E/(2*(1 + nu));
10
11A = geom(1);
12Izz = geom(2);
13Iyy = geom(3);
14ky = geom(4);
15J = geom(5);
16
17% Calculate entries of stiffness matrix
18X = E*A/L;
19S = G*J/L;
20
21phi_y = 12*E*Izz*ky/(A*G*L^2);
22Y1 = 12*E*Izz/((1 + phi_y)*L^3);
23Y2 = 6*E*Izz/((1 + phi_y)*L^2);
24Y3 = (4 + phi_y)*E*Izz/((1 + phi_y)*L);
25Y4 = (2 - phi_y)*E*Izz/((1 + phi_y)*L);
26
27phi_z = 12*E*Iyy*ky/(A*G*L^2);
28Z1 = 12*E*Iyy/((1 + phi_z)*L^3);
29Z2 = 6*E*Iyy/((1 + phi_z)*L^2);
30Z3 = (4 + phi_z)*E*Iyy/((1 + phi_z)*L);
31Z4 = (2 - phi_z)*E*Iyy/((1 + phi_z)*L);
32
33% Assemble matrix
34k = [X  0   0   0   0   0   -X  0   0   0   0   0
35    0   Y1  0   0   0   Y2  0   -Y1 0   0   0   Y2
36    0   0   Z1  0   -Z2 0   0   0   -Z1 0   -Z2 0
37    0   0   0   S   0   0   0   0   0   -S  0   0
38    0   0   -Z2 0   Z3  0   0   0   Z2  0   Z4  0
39    0   Y2  0   0   0   Y3  0   -Y2 0   0   0   Y4
40    -X  0   0   0   0   0   X   0   0   0   0   0
41    0   -Y1 0   0   0   -Y2 0   Y1  0   0   0   -Y2
42    0   0   -Z1 0   Z2  0   0   0   Z1  0   Z2  0
43    0   0   0   -S  0   0   0   0   0   S   0   0
44    0   0   -Z2 0   Z4  0   0   0   Z2  0   Z3  0
45    0   Y2  0   0   0   Y4  0   -Y2 0   0   0   Y3];
46
47end
