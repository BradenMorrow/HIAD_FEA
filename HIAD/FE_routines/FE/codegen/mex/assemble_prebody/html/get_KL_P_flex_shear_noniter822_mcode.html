1function [KL,P,el_out] = get_KL_P_flex_shear_noniter(el_in)
  2%GET_KL_P_FLEX
  3%   Calculate the element stiffness matrix and element forces using the
  4%   flexibility method.  Following Taucer et al. (1991), A Fiber
  5%   Beam-Column Element...
  6%   Shear deformability from Marini and Spacone (2006), Analysis of
  7%   Reinforced Concrete Elemnents Including Shear Effects
  8
  9% Extract variables
 10L = el_in.L; % Element length
 11R = el_in.r; % Radius of tube
 12cords = el_in.nodes; % Fiber constitutive relationships
 13p = el_in.p; % Internal pressure
 14beta = el_in.beta; % Braid angle
 15
 16EL = el_in.propsLH(1); % Longitudinal modulus
 17GLH = el_in.propsLH(3); % Inplane shear modulus
 18t = 3*el_in.propsLH(5); % Shell thickness
 19
 20A = 2*pi*R*t; % Shell area
 21I = pi*R^3*t; % Shell moment of inertia
 22
 23% State determination variables
 24K = el_in.flex.K; % Element stiffness
 25D = el_in.flex.D; % Section forces
 26Du = el_in.flex.Du; % Section forces
 27Q = el_in.flex.Q; % Element forces
 28f = el_in.flex.f; % Section compliance matrices
 29d = el_in.flex.d; % Section deformations
 30e = el_in.flex.e; % Section fiber strains
 31
 32% Number of integration points
 33n = el_in.n;
 34
 35% Initialize
 36alpha = el_in.alpha; % Location of fibers
 37l = zeros(size(alpha,1),3,n);
 38for i = 1:n
 39    l(:,:,i) = [ones(size(alpha,1),1) -R*cosd(alpha(:,i)) R*sind(alpha(:,i))]; % Location of fibers (cartesian)
 40end
 41
 42% Initialize
 43b = zeros(5,5,n); % Force interpolation functions
 44dD = zeros(5,n); % Section force increment
 45dd = zeros(5,n); % Section deformation increment
 46rho = zeros(5,n); % Residual deformations
 47de = zeros(size(l,1),n); % Fiber strain increments
 48
 49sig = zeros(size(l,1),n); % Fiber force
 50EA = zeros(size(l,1),n); % Fiber stiffness
 51k = zeros(5,5,n); % Section stiffness matrix
 52Dr = zeros(5,n); % Section resisting forces
 53% Du = zeros(5,n); % Section unbalanced forces
 54Fi = zeros(5,5,n); % Element flexibility matrix at section
 55F = zeros(5); % Element flexibility matrix
 56r = zeros(5,1); % Residual element deformations
 57% q = zeros(5,1); % Residual element deformations
 58
 59% Gauss-Lobatto integration constants
 60[xi,wi] = get_GL(n);
 61
 62dq = el_in.dD(1:5); % Taucer notation
 63
 64% Fiber beam element, non-iterative state determination, Neuenhofer and Filippou (1997)
 65% (1) Compute the nodal force increments
 66dQ = K*dq; % Change in element forces
 67
 68% Loop through sections
 69for i = 1:n
 70    x = L/2*xi(i) + L/2;
 71    
 72    % Force interpolation at sections
 73    b(:,:,i) = [1 0 0 0 0 % [axial Mz My Vz Vy]'
 74        0 (x/L - 1) (x/L) 0 0
 75        0 0 0 (x/L - 1) (x/L)
 76        0 -1/L -1/L 0 0
 77        0 0 0 1/L 1/L];
 78    
 79    % (2) Compute the section force increments
 80    % Section force increment
 81    dD(:,i) = b(:,:,i)*dQ + Du(:,i);
 82    
 83    % (3) Compute the section deformation increments
 84    dd(:,i) = f(:,:,i)*dD(:,i); % + r(:,i);
 85    d(:,i) = d(:,i) + dd(:,i);
 86    
 87    % (4) Compute section resisting forces
 88    % Compute the fiber deformation increments
 89    de(:,i) = l(:,:,i)*dd(1:3,i);
 90    e(:,i) = e(:,i) + de(:,i);
 91    
 92    % Compute fiber stresses and tangent moduli
 93    % Determine current fiber force and stiffness
 94    interp_meth = 'linear'; % 'spline'; % 'pchip'; %
 95    tol1 = 1e-10;
 96    for m = 1:size(alpha,1) % Fibers
 97        % Extract cord force - strain relationship
 98        axial = cords(i).cords(m).axial;
 99        
100        % Calculate cord stiffness
101        if(strcmp(interp_meth, 'linear'))
102            force = interp1qr(axial(:,2),axial(:,1),[e(m,i) e(m,i) + tol1]');
103        else
104            force = interp1(axial(:,2),axial(:,1),[e(m,i) e(m,i) + tol1]',interp_meth,'extrap');
105        end
106        sig(m,i) = force(1);
107        
108        % force_up = interp1(axial(:,2),axial(:,1),e(m,i) + tol1,interp_meth);
109        EA(m,i) = (force(2) - force(1))/tol1;
110        
111    end
112    
113    % (6) Compute the section tangent stiffness matrix
114    % Tangent stiffness
115    k(:,:,i) = zeros(5);
116    k(1:3,1:3,i) = l(:,:,i)'*diag(EA(:,i))*l(:,:,i);
117    
118    % Including elastic shell, with work done by pressure due to shear deformation induced volume change
119    k_shell = diag([EL*A EL*I EL*I (GLH*A/2 + p*pi*R^2) (GLH*A/2 + p*pi*R^2)]');
120    k(:,:,i) = k(:,:,i) + k_shell;
121    
122    % (7) Section tangent flexibility matrix
123    f(:,:,i) = k(:,:,i)\eye(5);
124    
125    % Reordered DOFs [axial Mz My Vz Vy]'
126    Dr(:,i) = [sum(sig(:,i))  + p*pi*R^2*2*cotd(beta)^2 + EL*A*d(1,i) - p*pi*R^2
127        sum(sig(:,i).*l(:,2,i))
128        sum(sig(:,i).*l(:,3,i))
129        0
130        0] + k_shell*d(:,i).*[0 1 1 1 1]'; % Includes axial prestress and shell forces
131    
132    % (8) Compute the element flexibility matrix
133    % Flexibility matrix at section
134    Fi(:,:,i) = b(:,:,i)'*f(:,:,i)*b(:,:,i);
135    
136    % Integrate b(x)'*f(x)*b(x) over the element length to obtain the element
137    % flexibility matrix, sum products of function evaluations and weights
138    F = F + L/2*wi(i)*Fi(:,:,i);
139    
140    % (9) Compute the element stiffness matrix - BELOW
141    
142    % (10) Compute the residual section deformations
143    rho(:,i) = f(:,:,i)*(D(:,i) + dD(:,i) - Dr(:,i));
144    
145    
146    % (11) Compute the residual element displacements
147    ri = b(:,:,i)'*rho(:,i);
148    r = r + L/2*wi(i)*ri;
149end
150
151% (9) Compute the element stiffness matrix
152% Inverse of the element flexibility matrix is the element stiffness
153% matrix, K = F^-1
154K = F\eye(5);
155
156% (12) Element resisting forces
157Q = Q + dQ - K*r; % New element forces
158
159% (13) Unbalanced section forces
160for i = 1:n
161    Du(:,i) = b(:,:,i)*Q - Dr(:,i);
162end
163
164
165
166
167% For passing back to corotational formulation
168J = 2*pi*R^3*t;
169
170% Element forces for corotational formulation (including linear torsion)
171P = [Q; el_in.D(6)*GLH*J/L];
172
173% Element stiffness matrix for corotational formulation (including linear
174% torsion)
175KL = zeros(6);
176KL(6,6) = GLH*J/L;
177KL(1:5,1:5) = K;
178
179
180
181
182
183% Store variables for future iterations
184el_out.break = 0;
185el_out.K = K; % Element stiffness
186el_out.D = Dr; % Section forces
187el_out.Du = Du; % Unbalanced section forces
188el_out.Q = Q; % Element forces
189el_out.f = f; % Section compliance matrices
190el_out.d = d; % Section deformations
191el_out.e = e; % Section fiber strains
192
193% % % % Plot strain at end of beam
194% % % if el_in.el == 3;
195% % %     plot_strain
196% % % end
197% % %
198% % % % For tracking change in volume
199% % % dV_L = L/2*wi'*(pi*R^2*d(1,:)');
200% % % dV_H = L/2*wi'*(pi*R^2*d(1,:)*nuLH.*(d(1,:)*nuLH - 2))';
201% % % dV = pi*R^2*(1 - 2*nuLH)*L/2*wi'*d(1,:)';
202% % %
203% % % el_out.dV_L = dV_L;
204% % % el_out.dV_H = dV_H;
205% % % el_out.dV = dV;
206
207end
