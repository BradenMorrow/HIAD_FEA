1function yi = interp1qr( x, y, xi )
 2%Quicker 1D linear interpolation with extrapolation in the positive x
 3%direction
 4% Performs 1D linear interpolation of 'xi' points using 'x' and 'y', resulting
 5% in 'yi' following the formula yi = y1 + y1 (y2-y1)/(x2-x1)*xi-x1).
 6% Returns NaN when 'xi' is Nan.
 7% 'x'  is column vector [m x 1], monotonically increasing.
 8% 'y'  is matrix [m x n], corresponding to 'x'.
 9% 'xi' is column vector [p x 1], in any order.
10% 'yi' is matrix [p x n], corresponding to 'xi'.
11
12% Size of 'x' and 'y'
13m = size(x,1);
14n = size(y,2);
15
16% For each 'xi', get the position of the 'x' element bounding it on the left [p x 1]
17[~,xi_pos] = histc(xi,x);
18xi_pos = max(xi_pos,1);     % To avoid index=0 when xi < x(1)
19xi_pos = min(xi_pos,m-1);   % To avoid index=m+1 when xi > x(end).
20
21% 't' matrix [p x 1]
22dxi = xi-x(xi_pos);
23dx = x(xi_pos+1)-x(xi_pos);
24t = dxi./dx;
25
26% Get 'yi'
27yi = y(xi_pos,:) + t(:,ones(1,n)).*(y(xi_pos+1,:)-y(xi_pos,:));
28
29% Give extrapolate the values of 'yi' corresponding to 'xi' out of the range of 'x'
30if (all(xi>x(end)))
31    delta_x = x(end)-x(end-1);
32    delta_y = y(end)-y(end-1);
33    extrap_val = xi>x(end);
34    yi(extrap_val,:) = delta_y*(xi(extrap_val)-x(end))/delta_x+y(end);
35end
36if (all(xi<x(1)))
37    delta_x = x(2)-x(1);
38    delta_y = y(2)-y(1);
39    extrap_val = xi<x(1);
40    yi(extrap_val,:) = -delta_y*(x(1)-xi(extrap_val))/delta_x+y(1);
41end
42
43end
