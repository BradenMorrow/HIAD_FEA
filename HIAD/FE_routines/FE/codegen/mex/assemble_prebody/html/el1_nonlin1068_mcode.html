1function [Kel,fint_i,Fint_i,ROT,el_out] = el1_nonlin(U_in,el_in,el_in0)
 2%EL1
 3%   Axial lookup, 3D beam element
 4
 5% Extract variables
 6% Geometry
 7nodes = el_in.nodes_ij;
 8orient = el_in.orient_ij;
 9
10% Materials
11mat = el_in0.mat;
12L0 = sum((nodes(2,:) - nodes(1,:)).^2).^.5;
13
14axial = el_in0.axial;
15axial_k = el_in0.axial_k;
16
17% Obtain displacements of interest
18Uel = U_in.U';
19Uel = Uel(:);
20
21% Transformation matrix
22T = T_beam_3d(nodes,orient);
23
24% Local element displacements
25u = T*Uel;
26
27% Axial strain
28eps = (u(7) - u(1))/L0;
29
30% Nonlinear axial response geometry
31f = ppval_fast(axial,eps); % Interpolate for force % 
32EAax = ppval_fast(axial_k,eps); % Stiffness is derivative of force-stain relationship
33
34geom = el_in0.geom;
35geom(1) = EAax/mat(1); % Vary cross sectional area
36
37% Local element stiffness matrix
38k = k1(mat,geom,L0);
39
40% Element stiffness matrix
41Kel = T'*k*T;
42
43% Element internal forces
44fint_i = k*u; % Local
45fint_i([1 7]) = [-f f];
46
47Fint_i = T'*fint_i; % Global
48
49% Extract rotations
50drot = U_in.delta_U(:,4:6);
51roti = U_in.U(:,4:6);
52ROT.rot = roti + drot; % Total rotation
53ROT.DELTA_rot = zeros(size(roti)); % Initialize ROT.DELTA_rot
54
55rot0 = U_in.U0(:,4:6);
56ROT.DELTA_rot = ROT.rot - rot0; % Increment change in rotation
57
58% Do not need to save any variables for future iterations
59el_out = el_in0;
60
61
62el_out.break = 0;
63
64end
