 40Sr1(2,3) = -r1(1);

 62g12 = -1/4*(A*z*e1'*Srk + (e1 + r1)'*z*A*Srk + A*rk*z'*Sr1); %3x3
 63g22 = 1/8*(-(rk'*e1)*Sz*Sr1 + Srk*e1*z'*Sr1 + 2*Sz*Srk + ...
