1function [PHI] = get_PHI_quat(qi)
 2%GET_PHI_QUAT
 3% Get a rotational vector from a quaternion
 4% de Souza (2000), figure 3.3
 5
 6q0 = qi(1);
 7
 8if q0 < 0
 9    qi = -qi;
10end
11
12q0 = qi(1);
13q = qi(2:4);
14
15if norm(q) == 0
16    PHI = [0 0 0]';
17    
18elseif norm(q) < q0
19    PHI = 2*q/norm(q)*asin(norm(q));
20    
21else
22    PHI = 2*q/norm(q)*acos(q0);
23end
24
25
26end
